**Step 2 — Create lib/config/regions.ts (GREEN):**

```typescript
export type RegionCode = 'ca' | 'uk' | 'us';

export interface RegionConfig {
  currency: 'CAD' | 'GBP' | 'USD';
  currencySymbol: string;
  locale: string;
  paymentMethod: string;
  countryCode: string;
  flagEmoji: string;
}

export const VALID_REGIONS: RegionCode[] = ['ca', 'uk', 'us'];

export const REGIONS: Record<RegionCode, RegionConfig> = {
  ca: {
    currency: 'CAD',
    currencySymbol: 'CA$',
    locale: 'en-CA',
    paymentMethod: 'Gift Card',
    countryCode: 'CA',
    flagEmoji: '🇨🇦',
  },
  uk: {
    currency: 'GBP',
    currencySymbol: '£',
    locale: 'en-GB',
    paymentMethod: 'WorldPay',
    countryCode: 'GB',
    flagEmoji: '🇬🇧',
  },
  us: {
    currency: 'USD',
    currencySymbol: '$',
    locale: 'en-US',
    paymentMethod: 'Gift Card',
    countryCode: 'US',
    flagEmoji: '🇺🇸',
  },
};

export function getRegionConfig(region: string): RegionConfig {
  if (!VALID_REGIONS.includes(region as RegionCode)) {
    throw new Error(`Unknown region: ${region}`);
  }
  return REGIONS[region as RegionCode];
}
```

**Step 3 — Update middleware.ts:**

Prepend region logic BEFORE the existing protected/admin route guards. Preserve all existing session/auth logic exactly:

```typescript
import { VALID_REGIONS } from '@/lib/config/regions';

// At the top of the middleware function, before any existing logic:
const { pathname } = request.nextUrl;

// Region handling: root redirect
if (pathname === '/') {
  return NextResponse.redirect(new URL('/uk', request.url));
}

// Region handling: extract segment, validate, redirect invalid to /uk
const pathSegments = pathname.split('/').filter(Boolean);
const maybeRegion = pathSegments[0];
const isRegionRoute = VALID_REGIONS.includes(maybeRegion as any);
const isNonRegionRoute = ['api', '_next', 'login', 'register', 'admin', 'account', 'dashboard'].includes(maybeRegion);

if (!isRegionRoute && !isNonRegionRoute && maybeRegion) {
  return NextResponse.redirect(new URL('/uk', request.url));
}

// Set x-region header for server components
if (isRegionRoute) {
  const requestHeaders = new Headers(request.headers);
  requestHeaders.set('x-region', maybeRegion);
  res = NextResponse.next({ request: { headers: requestHeaders } });
}

// ... existing session/auth logic follows unchanged ...
```

import { createContext, useContext } from 'react';
import type { RegionContextValue } from './regions';
export { type RegionContextValue };
export const RegionContext = createContext<RegionContextValue | null>(null);
export function useRegion(): RegionContextValue {
  const ctx = useContext(RegionContext);
  if (!ctx) throw new Error('useRegion must be used within RegionProvider');
  return ctx;
}
